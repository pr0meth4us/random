ឯកឧត្តមអគ្គនាយក នៃអគ្គនាយកដ្ឋានគោលនយោបាយ

កម្មវត្ថុ៖ ការផ្តល់ព័ត៌មានសេវាសាធារណៈ និងការចាត់តាំងជនបង្គោល ដើម្បីគាំទ្រការអនុវត្តសាកល្បងដំណើរការប្រព័ន្ធទំនាក់ទំនងអតិថិជនតាមទូរស័ព្ទជាមួយប្រព័ន្ធឆ្លើយតបស្វ័យប្រវត្តសម្រាប់សហគ្រាសធុនមីក្រូ តូច និងមធ្យម របស់ក្រសួងឧស្សាហកម្ម វិទ្យាសាស្ត្រ បច្ចេកវិទ្យា និងនវានុវត្តន៍។

យោង៖ លិខិតលេខ ២៩៣៧ / អគន ចុះថ្ងៃទី២០ ខែមិថុនា ឆ្នាំ២០២៣ របស់អគ្គនាយកដ្ឋានគោលនយោបាយ។

ជូនភ្ជាប់៖ ទម្រង់ប្រមូលព័ត៌មានសេវាសាធារណៈពាក់ព័ន្ធនឹងសហគ្រាសធុនមីក្រូ តូច និងមធ្យម របស់មជ្ឈមណ្ឌលបណ្តុះធុរកិច្ចថ្មី "តេជោ" ចំនួន ០៣ ច្បាប់។

សេចក្តីដូចមានចែងក្នុងកម្មវត្ថុ និងយោងខាងលើ ខ្ញុំសូមគោរពជម្រាបជូន ឯកឧត្តមអគ្គនាយក មេត្តាជ្រាបដ៏ខ្ពង់ខ្ពស់ថា៖ តបតាមការស្នើសុំ មជ្ឈមណ្ឌលបណ្តុះធុរកិច្ចថ្មី "តេជោ" បានរៀបចំបំពេញទម្រង់ព័ត៌មានសេវាសាធារណៈព្រមទាំងការចាត់តាំងសមាសភាពជនបង្គោល ក្រោមកម្មវិធីបរិវត្តកម្មឌីជីថលសហគ្រាស រួមមានសេវា៖ (១) វគ្គបណ្តុះបណ្តាលបរិវត្តកម្មឌីជីថល (Digital Adoption Training) (២) ការសិក្សាតាមប្រព័ន្ធអេឡិចត្រូនិក (E-Learning) និង (៣) បណ្ណជំនួយថវិកា (Grant Voucher) សម្រាប់គាំទ្រដល់ប្រព័ន្ធទំនាក់ទំនងអតិថិជនរបស់ក្រសួងឧស្សាហកម្ម វិទ្យាសាស្ត្រ បច្ចេកវិទ្យា និងនវានុវត្តន៍ រួចរាល់ដូចមានជូនភ្ជាប់មកជាមួយ។

សេចក្តីដូចបានគោរពជម្រាបជូនខាងលើ សូម ឯកឧត្តមអគ្គនាយក មេត្តាជ្រាប និងចាត់ចែងតាមការគួរ។
